        // Функция для форматирования даты

        function formatDate(dateString) {

            const options = { day: 'numeric', month: 'long', weekday: 'short' };

            const date = new Date(dateString);

            return date.toLocaleDateString('ru-RU', options);

        }

 

        // Функция для отображения периода

        function formatPeriod(startDate, endDate) {

            if (!endDate) return formatDate(startDate);

           

            const start = new Date(startDate);

            const end = new Date(endDate);

           

            if (start.getMonth() === end.getMonth()) {

                return `${start.getDate()}-${end.getDate()} ${start.toLocaleDateString('ru-RU', { month: 'long' })}`;

            } else {

                return `${start.getDate()} ${start.toLocaleDateString('ru-RU', { month: 'short' })} - ${end.getDate()} ${end.toLocaleDateString('ru-RU', { month: 'long' })}`;

            }

        }
